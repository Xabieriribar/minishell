--- 1. Basic Commands & Spaces ---
01: [ls]                      -> WORD(ls)
02: [ls -la]                  -> WORD(ls), WORD(-la)
03: [echo hello world]        -> WORD(echo), WORD(hello), WORD(world)
04: [   ls -l   ]             -> WORD(ls), WORD(-l)
05: [ls       -la]            -> WORD(ls), WORD(-la)
06: [/bin/ls]                 -> WORD(/bin/ls)
07: [./minishell]             -> WORD(./minishell)
08: [echo ""]                 -> WORD(echo), WORD()
09: [echo " "]                -> WORD(echo), WORD( )
10: ["" ""]                   -> WORD(), WORD()

--- 2. Single Quotes (Literal) ---
11: ['ls -la']                -> WORD(ls -la)
12: [echo '   hello   ']      -> WORD(echo), WORD(   hello   )
13: [echo '$USER']            -> WORD(echo), WORD($USER)
14: [echo 'quote "in" quote'] -> WORD(echo), WORD(quote "in" quote)
15: ['']                      -> WORD()
16: [' ']                     -> WORD( )
17: [cat 'file name']         -> WORD(cat), WORD(file name)

--- 3. Double Quotes ---
18: ["ls -la"]                -> WORD(ls -la)
19: [echo "hello $USER"]      -> WORD(echo), WORD(hello $USER)
20: [echo "'single quotes'"]  -> WORD(echo), WORD('single quotes')
21: ["      "]                -> WORD(      )
22: [echo "pipe | in quotes"] -> WORD(echo), WORD(pipe | in quotes)
23: [echo "> redir in quotes"]-> WORD(echo), WORD(> redir in quotes)
24: [echo "$?"]               -> WORD(echo), WORD($?)

--- 4. Token Concatenation (Glue) ---
25: [echo"hello"]             -> WORD(echohello)
26: [ls"-la"]                 -> WORD(ls-la)
27: [echo'hello'world]        -> WORD(echohelloworld)
28: [cat"file"'name']         -> WORD(catfilename)
29: [""ls]                    -> WORD(ls)
30: [ls""]                    -> WORD(ls)

--- 5. Simple Redirections & Pipes ---
31: [ls|wc]                   -> WORD(ls), PIPE, WORD(wc)
32: [ls | wc]                 -> WORD(ls), PIPE, WORD(wc)
33: [cat<file]                -> WORD(cat), RED_IN, WORD(file)
34: [ls>file]                 -> WORD(ls), RED_OUT, WORD(file)
35: [ls>>file]                -> WORD(ls), APPEND, WORD(file)
36: [cat<<limit]              -> WORD(cat), HEREDOC, WORD(limit)
37: [ls | wc | grep 0]        -> WORD(ls), PIPE, WORD(wc), PIPE, WORD(grep), WORD(0)
38: [ls|wc|grep|awk]          -> WORD(ls), PIPE, WORD(wc), PIPE, WORD(grep), PIPE, WORD(awk)

--- 6. Mixed Operators & Commands ---
39: [ls -l | grep .c > out]   -> WORD(ls), WORD(-l), PIPE, WORD(grep), WORD(.c), RED_OUT, WORD(out)
40: [cat < in | wc -l > out]  -> WORD(cat), RED_IN, WORD(in), PIPE, WORD(wc), WORD(-l), RED_OUT, WORD(out)
41: [ls > out | wc -l]        -> WORD(ls), RED_OUT, WORD(out), PIPE, WORD(wc), WORD(-l)
42: [> out ls]                -> RED_OUT, WORD(out), WORD(ls)
43: [ls -l > out1 -a > out2]  -> WORD(ls), WORD(-l), RED_OUT, WORD(out1), WORD(-a), RED_OUT, WORD(out2)
44: [cat < in | grep a >> out | wc -l] -> WORD(cat), RED_IN, WORD(in), PIPE, WORD(grep), WORD(a), APPEND, WORD(out), PIPE, WORD(wc), WORD(-l)

--- 7. The Evil Tests ---
45: [ls | | wc]               -> WORD(ls), PIPE, PIPE, WORD(wc)
46: [ls > > out]              -> WORD(ls), RED_OUT, RED_OUT, WORD(out)
47: [| ls]                    -> PIPE, WORD(ls)
48: [ls |]                    -> WORD(ls), PIPE
49: [ls >]                    -> WORD(ls), RED_OUT
50: ["" | '' > " "]           -> WORD(), PIPE, WORD(), RED_OUT, WORD( )
